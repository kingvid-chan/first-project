②ios手机不支持音频自动播放，经典解决方式是通过引导用户发生触摸事件来使音频播放。
<audio loop style="display:none" id="audio">       //loop
        <source src="./song.ogg" type="audio/ogg">
        <source src="./song.mp3" type="audio/mpeg" />
</audio>
var myaudio=document.getElementById("audio");
$("***").one("touchstart", function() {    //第一次触屏事件触发音乐播放
        myaudio.play();
});
